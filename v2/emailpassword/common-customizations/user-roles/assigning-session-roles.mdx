# Assigning roles to a session

This can be done at two points in time:
- 1) During user login / sign up
- 2) In any API call post login

## 1) During user login / sign up

We can set the user's role in the access token by overriding the `createNewSession` function in the `init` function:

<BackendSDKTabs>
<TabItem value="nodejs">

```jsx
let SuperTokens = require("supertokens-node");
let Session = require("supertokens-node/recipe/session");

SuperTokens.init({
    SuperTokens: {...},
    appInfo: {...},
    recipeList: [
        Session.init({
            // highlight-start
            override: {
                functions: (originalImplementation) => {
                    return {
                        ...originalImplementation,
                        createNewSession: async (input) => {
                            let userId = input.userId;

                            let role = "admin"; // TODO: fetch role based on userId

                            input.jwtPayload = {
                                ...input.jwtPayload,
                                role
                            };

                            return originalImplementation.createNewSession(input);
                        },
                    };
                },
            },
            // highlight-end
        })
    ]
});
```
</TabItem>
<TabItem value="go">

```go
import (
	"net/http"

	"github.com/supertokens/supertokens-golang/recipe/session"
	"github.com/supertokens/supertokens-golang/recipe/session/sessmodels"
	"github.com/supertokens/supertokens-golang/supertokens"
)

func main() {

	supertokens.Init(supertokens.TypeInput{
		RecipeList: []supertokens.Recipe{
			session.Init(&sessmodels.TypeInput{
				Override: &sessmodels.OverrideStruct{
					Functions: func(originalImplementation sessmodels.RecipeInterface) sessmodels.RecipeInterface {
						return sessmodels.RecipeInterface{
							//highlight-start
							CreateNewSession: func(res http.ResponseWriter, userID string, jwtPayload, sessionData interface{}) (sessmodels.SessionContainer, error) {
								role := "admin" // TODO: fetch role based on userID

								if jwtPayload == nil {
									jwtPayload = map[string]interface{}{}
								}
								(jwtPayload.(map[string]interface{}))["role"] = role

								return originalImplementation.CreateNewSession(res, userID, jwtPayload, sessionData)
							},
							//highlight-end

							// other session functions are unmodified
							GetSession:                  originalImplementation.GetSession,
							RefreshSession:              originalImplementation.RefreshSession,
							GetSessionInformation:       originalImplementation.GetSessionInformation,
							RevokeAllSessionsForUser:    originalImplementation.RevokeAllSessionsForUser,
							GetAllSessionHandlesForUser: originalImplementation.GetAllSessionHandlesForUser,
							RevokeSession:               originalImplementation.RevokeSession,
							RevokeMultipleSessions:      originalImplementation.RevokeMultipleSessions,
							UpdateSessionData:           originalImplementation.UpdateSessionData,
							GetAccessTokenLifeTimeMS:    originalImplementation.GetAccessTokenLifeTimeMS,
							GetRefreshTokenLifeTimeMS:   originalImplementation.GetRefreshTokenLifeTimeMS,
							UpdateJWTPayload:            originalImplementation.UpdateJWTPayload,
						}
					},
				},
			}),
		},
	})
}
```

## 2) In any API call post login

Post session verification, you can use the `updateJWTPayload` function to store the user's role:

<BackendSDKTabs>
<TabItem value="nodejs">
<NodeJSFrameworkSubTabs>
<TabItem value="express">

```jsx
let { verifySession } = require("supertokens-node/recipe/session/framework/express");

app.post("/set-role", verifySession(), async (req, res) => {

    let userId = req.session.getUserId(); 

    let role = "admin"; // TODO: fetch based on user

    // Note that this will override any existing JWT payload
    // that you may have provided earlier.
    // highlight-start
    await req.session.updateJWTPayload(
        {role}
    );
    // highlight-end

    //....
});
```
