# Provide SMTP configuration

Using this method, you can:
- Send emails using your own domain
- Optionally customise the default email templates and subject.

<BackendSDKTabs>
<TabItem value="nodejs">

```tsx
import supertokens from "supertokens-node";
import ^{recipeNameCapitalLetters} from "supertokens-node/recipe/^{codeImportRecipeName}";
import Session from "supertokens-node/recipe/session";
import { STMPService } from "supertokens-node/recipe/^{codeImportRecipeName}/emaildelivery";

supertokens.init({
    appInfo: {
        apiDomain: "...",
        appName: "...",
        websiteDomain: "..."
    },
    recipeList: [
        ^{recipeNameCapitalLetters}.init({
            ^{nodeRecipeInitDefault}
            // highlight-start
            emailDelivery: {
                service: new STMPService({
                    smtpSettings: {
                        host: "...",
                        password: "...",
                        port: 465,
                        from: {
                            name: "...",
                            email: "...",
                        },
                        secure: true
                    },
                })
            },
            // highlight-end
        }),
        Session.init()
    ]
});
```
</TabItem>
<TabItem value="go">

```go
import (
    "github.com/supertokens/supertokens-golang/ingredients/emaildelivery"
    "github.com/supertokens/supertokens-golang/recipe/^{codeImportRecipeName}"
    "github.com/supertokens/supertokens-golang/recipe/^{codeImportRecipeName}/emaildelivery/smtpService"
    "github.com/supertokens/supertokens-golang/recipe/^{codeImportRecipeName}/^{goModelName}"
    "github.com/supertokens/supertokens-golang/supertokens"
)

func main() {
    smtpSecure := false
    // highlight-start
    smtpService := smtpService.MakeSmtpService(emaildelivery.SMTPTypeInput{
		SMTPSettings: emaildelivery.SMTPServiceConfig{
			Host: "...",
			From: emaildelivery.SMTPServiceFromConfig{
				Name:  "...",
				Email: "...",
			},
			Port:     456,
			Password: "...",
			Secure:   &smtpSecure,
		},
	})
    // highlight-end
    supertokens.Init(supertokens.TypeInput{
        RecipeList: []supertokens.Recipe{
            ^{codeImportRecipeName}.Init(^{goModelNameForInit}.TypeInput{
                ^{goRecipeInitDefault} // typecheck-only, removed from output
                // highlight-start
                EmailDelivery: &emaildelivery.TypeInput{
                    Service: &smtpService,
                },
                // highlight-end
            }),
        },
    })
}
```

<TabItem value="python">

```python
from supertokens_python import init, InputAppInfo
from supertokens_python.recipe import ^{codeImportRecipeName}
from supertokens_python.ingredients.emaildelivery.services.smtp import SMTPServiceConfig, SMTPServiceConfigFrom, EmailDeliverySMTPConfig
from supertokens_python.ingredients.emaildelivery.types import EmailDeliveryConfig
from supertokens_python.recipe.^{codeImportRecipeName}.emaildelivery.services import smtp

init(
    app_info=InputAppInfo(api_domain="...", app_name="...", website_domain="..."),
    framework='...', # type: ignore
    recipe_list=[
        ^{codeImportRecipeName}.init(
            ^{pythonRecipeInitDefault} # typecheck-only, removed from output
            # highlight-start
            email_delivery=EmailDeliveryConfig(
                service=smtp.SMTPService(
                    config=EmailDeliverySMTPConfig(
                        smtp_settings=SMTPServiceConfig(
                            host="...",
                            port=465,
                            from_=SMTPServiceConfigFrom(
                                name="...",
                                email="..."
                            ),
                            password="...",
                            secure=False
                        )
                    )
                )
            )
            # highlight-end
        )
    ]
)
```
