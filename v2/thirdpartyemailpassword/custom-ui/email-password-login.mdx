```tsx
import { emailPasswordSignUp } from "supertokens-web-js/recipe/^{codeImportRecipeName}";
import SuperTokensGeneralError from "supertokens-web-js/utils/error"

async function signUpClicked(email: string, password: string, name: string) {
    try {
        let response = await emailPasswordSignUp({
            formFields: [{
                id: "email", // required field
                value: email
            }, {
                id: "password", // required field
                value: password
            },
            {
                id: "name", // custom field - the ID can be anything you like
                value: name
            }]
        })

        if (response.status === "FIELD_ERROR") {
            // one of the input formFields failed validaiton
            response.formFields.forEach(formField => {
                if (formField.id === "email") {
                    // Email validation failed (for example incorrect email syntax),
                    // or the email is not unique.
                    window.alert("Invalid email, or the email already exists")
                } else if (formField.id === "password") {
                    // Password validation failed.
                    // Maybe it didn't match the password strength
                    window.alert("Password is weak")
                } else if (formField.id === "name") {
                    // name field validation failed
                    window.alert("name input failed validation");
                }
            })
        } else {
            // sign up successful. The session tokens are automatically handled by
            // the frontend SDK.
            window.location.href = "/homepage"
        }
    } catch (err) {
        if (SuperTokensGeneralError.isThisError(err)) {
            // this may be a custom error message sent from the API by you.
            window.alert(err.message);
        } else {
            window.alert("Oops! Something went wrong.");
        }
    }
}
```
