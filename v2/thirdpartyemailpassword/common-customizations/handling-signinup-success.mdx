# Post signin / signup callbacks

## 1) On the frontend

This method allows you to fire events immediately after a successful sign in / up. For example to send analytics events post sign in / up.

<FrontendSDKTabs>
<TabItem value="reactjs">

```jsx
SuperTokens.init({
    appInfo: {...},
    recipeList: [
        ThirdPartyEmailPassword.init({
            // highlight-start
            onHandleEvent: async (context) => {
                if (context.action === "SESSION_ALREADY_EXISTS") {
                    // TODO:
                } else {
                    let {id, email} = context.user;
                    if (context.action === "SUCCESS") {
                        if (context.isNewUser) {
                            // TODO: Sign up
                        } else {
                            // TODO: Sign in
                        }
                    }
                }
            }
            // highlight-end
        }),
        Session.init()
    ]
});
```

:::info
Please refer to [this page](../advanced-customizations/frontend-hooks/handle-event) to learn more about the `onHandleEvent` hook.
:::

## 2) On the backend

For this, you'll have to override the `signInUpPOST` API in the `init` function call.

<BackendSDKTabs>
<TabItem value="nodejs">

```jsx
SuperTokens.init({
    appInfo: {...},
    supertokens: {...},
    recipeList: [
        ThirdPartyEmailPassword.init({
            // highlight-start
            override: {
                apis: (originalImplementation) => {
                    return {
                        ...originalImplementation,
                        signInUpPOST: async (input) => {

                            if (originalImplementation.signInUpPOST === undefined) {
                                throw Error("Should never come here");
                            }

                            // First we call the original implementation of signInUpPOST.
                            let response = await originalImplementation.signInUpPOST(input);

                            // Post sign up response, we check if it was successful
                            if (response.status === "OK") {

                                let { id, email } = response.user;

                                // Then we check if the user signed up using email / password or a third party login provider.
                                if (response.type === "thirdparty") {
                                    // This is the response from the OAuth 2 provider that contains their tokens or user info.
                                    let thirdPartyAuthCodeResponse = response.authCodeResponse;
                                }
                                if (input.type === "emailpassword") {
                                    // These are the input form fields values that the user used while signing up / in
                                    let formFields = input.formFields;
                                }

                                if (response.createdNewUser) {
                                    // TODO: Post sign up logic
                                } else {
                                    // TODO: Post sign in logic
                                }
                            }
                            return response;
                        }
                    }
                }
            }
            // highlight-end
        }),
        Session.init({...})
    ]
});
```

</TabItem>
<TabItem value="go">

```go
import (
	"github.com/supertokens/supertokens-golang/recipe/thirdpartyemailpassword"
	"github.com/supertokens/supertokens-golang/recipe/thirdpartyemailpassword/tpepmodels"
	"github.com/supertokens/supertokens-golang/supertokens"
)

func main() {

	supertokens.Init(supertokens.TypeInput{
		RecipeList: []supertokens.Recipe{
			thirdpartyemailpassword.Init(&tpepmodels.TypeInput{
                //highlight-start
				Override: &tpepmodels.OverrideStruct{
					APIs: func(originalImplementation tpepmodels.APIInterface) tpepmodels.APIInterface {
                        // create a copy of the originalImplementation
                        newImplementation := originalImplementation

                        // override the sign in up API
                        newImplementation.SignInUpPOST = func(input tpepmodels.SignInUpAPIInput) (tpepmodels.SignInUpAPIOutput, error) {

                            // First we call the original implementation of SignInUpPOST.
                            response, err := originalImplementation.SignInUpPOST(input)
                            if err != nil {
                                return tpepmodels.SignInUpAPIOutput{}, err
                            }

                            if response.EmailpasswordOutput != nil {
                                // User sign in or up using email password
                                if response.EmailpasswordOutput.OK != nil {
                                    // sign in / up was successful

                                    // user object contains the ID and email
                                    user := response.EmailpasswordOutput.OK.User

                                    // formFields contains the values in the form the user filled up during sign in / up
                                    formFields := input.EmailpasswordInput.FormFields

                                    if response.EmailpasswordOutput.OK.CreatedNewUser {
                                        // TODO: Post sign up logic
                                    } else {
                                        // TODO: Post sign in logic
                                    }

                                }
                            } else {
                                // User sign in or up using social login
                                if response.ThirdPartyOutput.OK != nil {
                                    // sign in / up was successful

                                    // user object contains the ID, email and other third party info
                                    user := response.ThirdPartyOutput.OK.User

                                    // This is the response from the OAuth 2 provider that contains their tokens or user info.
                                    thirdPartyAuthCodeResponse := response.ThirdPartyOutput.OK.AuthCodeResponse

                                    if response.ThirdPartyOutput.OK.CreatedNewUser {
                                        // TODO: Post sign up logic
                                    } else {
                                        // TODO: Post sign in logic
                                    }

                                }
                            }
                            return response, nil
                        }

                        // return the new implementation
						return newImplementation
					},
				},
                //highlight-end
			}),
		},
	})
}
```
