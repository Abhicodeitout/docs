# Manually generating a link

You can use our backend SDK to generate the reset password link as shown below:

<BackendSDKTabs>
<TabItem value="nodejs">

```tsx
import ^{recipeNameCapitalLetters} from "supertokens-node/recipe/^{codeImportRecipeName}";

async function createResetPasswordLink(userId: string) {
    const linkResponse = await ^{recipeNameCapitalLetters}.createResetPasswordLink(userId);

    if (linkResponse.status === "OK") {
        console.log(linkResponse.link);
    } else {
        // user does not exist or is not an email password user
    }
}
```

</TabItem>
<TabItem value="go">

```go
import (
    "github.com/supertokens/supertokens-golang/recipe/^{codeImportRecipeName}"
)

func main() {
    userID := "..."

    linkRes, err := ^{codeImportRecipeName}.CreateResetPasswordLink(userID)
    if err != nil {
        // handle error
    }

    if linkRes.OK != nil {
        link := linkRes.link
        fmt.Println(link)

    } else {
        // user does not exist or is not an email password user
    }
}
```

</TabItem>

<PythonSyncAsyncSubTabs>
<TabItem value="asyncio">

```python
from supertokens_python.recipe.^{codeImportRecipeName}.asyncio import create_reset_password_link
from supertokens_python.recipe.^{codeImportRecipeName}.interfaces import CreateResetPasswordTokenOkResult

async def create_link(user_id: str):
    link_res = await create_reset_password_link(user_id)

    if isinstance(token_res, CreateResetPasswordTokenOkResult):
        link = link_res.link
        print(link)
    else:
        # user does not exist or is not an email password user
```

<CustomAdmonition type="multi-tenancy">

For a multi tenant setup, you can pass in the user's tenantId to the `createResetPasswordLink` function as well. The generated link will use the configured `websiteDomain` from the `appInfo` object (in `supertokens.init`), however, you can change the domain of the generated link to match that of the tenant ID.
